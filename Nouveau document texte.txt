// Définir une liste vide pour stocker les tâches
tâches = []

// Fonction pour ajouter une tâche
fonction ajouterTâche(tâche) :
    // Créer un nouvel objet tâche
    nouvelleTâche = {
        nom: tâche,
        terminée: false
    }

    // Ajouter la nouvelle tâche à la liste
    tâches.ajouter(nouvelleTâche)

// Fonction pour marquer une tâche comme terminée
fonction marquerCommeTerminée(index) :
    tâche = tâches[index]
    tâche.terminée = vrai

// Fonction pour afficher la liste des tâches
fonction afficherListe() :
    pour chaque tâche dans tâches :
        si tâche.terminée :
            afficher "[X] " + tâche.nom
        sinon :
            afficher "[ ] " + tâche.nom

// Exemple d'utilisation
ajouterTâche("Faire les courses")
ajouterTâche("Répondre aux e-mails")
marquerCommeTerminée(0)
afficherListe()
cliquer sur la card d'une tâche pour ouvrir le radial menu et choisir une option
(tâche finie, surppimer, reporter, etc)
